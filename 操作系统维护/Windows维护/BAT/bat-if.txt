没有 elif 或者 else if

IF 条件 (
    条件为真的时候的处理
) else (
    条件为假的时候的处理
)

IF abcdef==abcdef (
    echo abcdef==abcdef
)

IF EXIST F:\tmp\a.bat (
    echo EXIST:F:\tmp\a.bat
)

EQU - 等于（等しい）
NEQ - 不等于（等しくない）
LSS - 小于（より小さい）
LEQ - 小于等于（以下）
GTR - 大于（より大きい）
GEQ - 大于等于（以上）

IF a EQU b (
    echo a EQU b
)

@echo off 
set "str=this is a test" 
REM 检测变量%str%是否等于test，如果相等，显示OK，否则显示NO 
if "%str%"=="test" (echo OK) else echo NO 
pause>nul

if defined str2 echo str2已经被定义
